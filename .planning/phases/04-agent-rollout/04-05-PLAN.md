    is_parent_agent = True  # Has children: proteomics_de_analysis_expert, biomarker_discovery_expert

    factory_name = "de_analysis_expert"
    is_parent_agent = False

    factory_name = "biomarker_discovery_expert"
    is_parent_agent = False
